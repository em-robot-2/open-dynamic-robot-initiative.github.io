270  void loop()
271  {
272  const double start_time =
273  real_time_tools::Timer::get_current_time_sec();
274 
275  // wait until first desired_action was received
276  // ----------------------------
277  while (!has_shutdown_request() &&
278  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
279  {
280  const double now = real_time_tools::Timer::get_current_time_sec();
281  if (now - start_time > first_action_timeout_)
282  {
283  Status status;
284  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
285  "First action was not provided in time");
286  termination_reason_ = TerminationReason::FIRST_ACTION_TIMEOUT;
287 
288  robot_data_->status->append(status);
289 
290  std::cerr << "Error: " << status.error_message
291  << "\nRobot is shut down." << std::endl;
292 
293  request_shutdown();
294  break;
295  }
296  }
297 
298  for (long int t = 0; !has_shutdown_request(); t++)
299  {
300  // TODO: figure out latency stuff!!
301 
302  Status status;
303 
304  if (max_number_of_actions_ > 0 && t >= max_number_of_actions_)
305  {
306  // TODO this is not really an error
307  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
308  "Maximum number of actions reached.");
309  termination_reason_ =
310  TerminationReason::MAXIMUM_NUMBER_OF_ACTIONS_REACHED;
311  }
312 
313  timer_.start();
314 
315  // get latest observation from robot and append it to robot_data_
316  Observation observation = robot_driver_->get_latest_observation();
317  timer_.checkpoint("get observation");
318 
319  robot_data_->observation->append(observation);
320  // TODO: for some reason this sometimes takes more than 2 ms
321  // i think this may be due to a non-realtime thread blocking the
322  // timeseries. this is in fact an issue, we might have to
323  // duplicate all the timeseries and have a realtime thread
324  // writing back and forth
325  timer_.checkpoint("append observation");
326 
327  // If real time mode is enabled the next action needs to be provided
328  // in time. If this is not the case, optionally repeat the previous
329  // action or raise an error.
330  if (real_time_mode_ &&
331  robot_data_->desired_action->newest_timeindex() < t)
332  {
333  uint32_t action_repetitions =
334  robot_data_->status->newest_element().action_repetitions;
335 
336  if (action_repetitions < max_action_repetitions_)
337  {
338  robot_data_->desired_action->append(
339  robot_data_->desired_action->newest_element());
340  status.action_repetitions = action_repetitions + 1;
341  }
342  else
343  {
344  // No action provided and number of allowed repetitions
345  // of the previous action is exceeded --> Error
346  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
347  "Next action was not provided in time");
348  termination_reason_ =
349  TerminationReason::NEXT_ACTION_TIMEOUT;
350  }
351  }
352 
353  std::string driver_error_msg = robot_driver_->get_error();
354  if (!driver_error_msg.empty())
355  {
356  status.set_error(Status::ErrorStatus::DRIVER_ERROR,
357  driver_error_msg);
358  termination_reason_ = TerminationReason::DRIVER_ERROR;
359  }
360 
361  robot_data_->status->append(status);
362 
363  // if there is an error, shut robot down and stop loop
364  if (status.error_status != Status::ErrorStatus::NO_ERROR)
365  {
366  std::cerr << "Error: " << status.error_message
367  << "\nRobot is shut down." << std::endl;
368  break;
369  }
370  timer_.checkpoint("status");
371 
372  // early exit if destructor has been called
373  while (!has_shutdown_request() &&
374  !robot_data_->desired_action->wait_for_timeindex(t, 0.1))
375  {
376  }
377  if (has_shutdown_request())
378  {
379  break;
380  }
381 
382  Action desired_action = (*robot_data_->desired_action)[t];
383  timer_.checkpoint("get action");
384 
385  Action applied_action = robot_driver_->apply_action(desired_action);
386  timer_.checkpoint("apply action");
387 
388  robot_data_->applied_action->append(applied_action);
389  timer_.checkpoint("append applied action");
390 
391  if (t % 5000 == 0 && t > 0)
392  {
393  timer_.print_statistics();
394  }
395  }
396 
397  robot_driver_->shutdown();
398 
399  // If no specific termination reason was set, assume that the shutdown
400  // was requested from outside.
401  if (termination_reason_ == TerminationReason::NOT_TERMINATED)
402  {
403  termination_reason_ = TerminationReason::SHUTDOWN_REQUESTED;
404  }
405 
406  loop_is_running_ = false;
407  }
