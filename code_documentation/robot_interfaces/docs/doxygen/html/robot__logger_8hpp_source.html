116  void stop()
117  {
118  // This is a bit complicated: In any case, join the thread if it is
119  // joinable. However, only write the remaining data to file if the
120  // logging thread is actually running at the moment stop() is called.
121  bool still_running = is_running_;
122 
123  stop_was_called_ = true;
124  if (thread_.joinable())
125  {
126  thread_.join();
127  }
128 
129  if (still_running)
130  {
131  append_robot_data_to_file(index_, block_size_);
132  }

160  void write_current_buffer(const std::string filename,
161  long int start_index = 0,
162  long int end_index = -1)
163  {
164  if (is_running_)
165  {
166  throw std::runtime_error(
167  "RobotLogger is currently running. Call stop() first.");
168  }
169 
170  if (logger_data_->observation->length() == 0)
171  {
172  std::cout
173  << "Warning: RobotLogger buffer is empty. Nothing to write."
174  << std::endl;
175  return;
176  }
177 
178  output_file_name_ = filename;
179  write_header_to_file();
180 
181  // set end_index to the current time index if not set or if it is in the
182  // future.
183  long int t = logger_data_->observation->newest_timeindex();
184  if (end_index < 0)
185  {
186  end_index = t;
187  }
188  else
189  {
190  end_index = std::min(t, end_index);
191  }
192 
193  append_robot_data_to_file(start_index, end_index - start_index);

196  void write_current_buffer_binary(const std::string filename,
197  long int start_index = 0,
198  long int end_index = -1)
199  {
200  if (is_running_)
201  {
202  throw std::runtime_error(
203  "RobotLogger is currently running. Call stop() first.");
204  }
205 
206  if (logger_data_->observation->length() == 0)
207  {
208  std::cout
209  << "Warning: RobotLogger buffer is empty. Nothing to write."
210  << std::endl;
211  return;
212  }
213 
214  // set end_index to the current time index if not set or if it is in the
215  // future.
216  long int t = logger_data_->observation->newest_timeindex();
217  if (end_index < 0)
218  {
219  end_index = t;
220  }
221  else
222  {
223  end_index = std::min(t, end_index);
224  }
225 
226  long int block_size = end_index - start_index;
227 
228  std::vector<LogEntry> log_data;
229  log_data.reserve(block_size);
230 
231  for (long int t = start_index;
232  t < std::min(start_index + block_size,
233  logger_data_->observation->newest_timeindex());
234  t++)
235  {
236  try
237  {
238  LogEntry entry;
239  entry.timeindex = t;
240  entry.applied_action = (*logger_data_->applied_action)[t];
241  entry.desired_action = (*logger_data_->desired_action)[t];
242  entry.observation = (*logger_data_->observation)[t];
243  entry.status = (*logger_data_->status)[t];
244  entry.timestamp = logger_data_->observation->timestamp_s(t);
245 
246  log_data.push_back(entry);
247  }
248  catch (const std::invalid_argument &e)
249  {
250  auto t_oldest = logger_data_->observation->oldest_timeindex();
251  auto diff = t_oldest - t;
252 
253  std::cout << "Warning: Trying to log time step " << t
254  << " which is not in the buffer anymore. Skip "
255  << diff << " steps to time step " << (t_oldest + 1)
256  << std::endl;
257 
258  // Note that t is incremented in the next iteration, so logging
259  // continues from t_oldest + 1. This means that one potentially
260  // available step is dropped but it lowers the risk that the
261  // same issue happens immediately again as t_oldest may drop out
262  // of the buffer until it is processed in the next iteration.
263  t = t_oldest;
264  }
265  }
266 
267  std::ofstream outfile(filename, std::ios::binary);
268  cereal::BinaryOutputArchive archive(outfile);
269 
270  // add version information to the output file (this can be used while
271  // loading when the data format changes
272  const std::uint32_t format_version = 2;
273 
274  archive(format_version, log_data);
275  }

338  void append_names_to_header(
339  const std::string &identifier,
340  const std::vector<std::string> &field_name,
341  const std::vector<std::vector<double>> &field_data,
342  std::vector<std::string> &header)
343  {
344  for (size_t i = 0; i < field_name.size(); i++)
345  {
346  if (field_data[i].size() == 1)
347  {
348  std::string temp = identifier + "_" + field_name[i];
349  header.push_back(temp);
350  }
351  else
352  {
353  for (size_t j = 0; j < field_data[i].size(); j++)
354  {
355  std::string temp = identifier + "_" + field_name[i] + "_" +
356  std::to_string(j);
357  header.push_back(temp);
358  }
359  }
360  }

385  void append_robot_data_to_file(long int start_index, long int block_size)
386  {
387  output_file_.open(output_file_name_, std::ios_base::app);
388  output_file_.precision(27);
389 
390  for (long int t = start_index;
391  t < std::min(start_index + block_size,
392  logger_data_->observation->newest_timeindex());
393  t++)
394  {
395  try
396  {
397  Action applied_action = (*logger_data_->applied_action)[t];
398  Action desired_action = (*logger_data_->desired_action)[t];
399  Observation observation = (*logger_data_->observation)[t];
400  Status status = (*logger_data_->status)[t];
401  auto timestamp = logger_data_->observation->timestamp_s(t);
402 
403  output_file_ << t << " " << timestamp << " ";
404  append_field_data_to_file(status.get_data());
405  append_field_data_to_file(observation.get_data());
406  append_field_data_to_file(applied_action.get_data());
407  append_field_data_to_file(desired_action.get_data());
408  output_file_ << std::endl;
409  }
410  catch (const std::invalid_argument &e)
411  {
412  auto t_oldest = logger_data_->observation->oldest_timeindex();
413  auto diff = t_oldest - t;
414 
415  std::cout << "Warning: Trying to log time step " << t
416  << " which is not in the buffer anymore. Skip "
417  << diff << " steps to time step " << (t_oldest + 1)
418  << std::endl;
419 
420  // Note that t is incremented in the next iteration, so logging
421  // continues from t_oldest + 1. This means that one potentially
422  // available step is dropped but it lowers the risk that the
423  // same issue happens immediately again as t_oldest may drop out
424  // of the buffer until it is processed in the next iteration.
425  t = t_oldest;
426  }
427  }
428 
429  output_file_.close();
430  }

455  void loop()
456  {
457  is_running_ = true;
458 
459  write_header_to_file();
460 
461  while (!stop_was_called_ &&
462  !(logger_data_->desired_action->length() > 0))
463  {
464  std::this_thread::sleep_for(std::chrono::milliseconds(100));
465  }
466 
467  index_ = logger_data_->observation->newest_timeindex();
468 
469  while (!stop_was_called_)
470  {
471  if (index_ + block_size_ <=
472  logger_data_->observation->newest_timeindex())
473  {
474 #ifdef VERBOSE
475  auto t1 = std::chrono::high_resolution_clock::now();
476 #endif
477 
478  append_robot_data_to_file(index_, block_size_);
479 
480  index_ += block_size_;
481 
482 #ifdef VERBOSE
483  auto t2 = std::chrono::high_resolution_clock::now();
484  auto duration =
485  std::chrono::duration_cast<std::chrono::microseconds>(t2 -
486  t1)
487  .count();
488 
489  // to print the time taken for one block of data to be logged.
490  std::cout << "Time taken for one block of data to be logged: "
491  << duration << std::endl;
492 #endif
493  }
494  }
495 
496  is_running_ = false;
497  }
