287 std::string NJBRD::get_error()
288 {
289  // Checks each board for errors and translates the error codes into
290  // human-readable strings. If multiple boards have errors, the messages
291  // are concatenated. Each message is prepended with the index of the
292  // corresponding board.
293 
294  std::string error_msg = "";
295 
296  for (size_t i = 0; i < motor_boards_.size(); i++)
297  {
298  auto status_timeseries = motor_boards_[i]->get_status();
299  if (status_timeseries->length() > 0)
300  {
301  std::string board_error_msg = "";
302  using ErrorCodes = blmc_drivers::MotorBoardStatus::ErrorCodes;
303  switch (status_timeseries->newest_element().error_code)
304  {
305  case ErrorCodes::NONE:
306  break;
307  case ErrorCodes::ENCODER:
308  board_error_msg = "Encoder Error";
309  break;
310  case ErrorCodes::CAN_RECV_TIMEOUT:
311  board_error_msg = "CAN Receive Timeout";
312  break;
313  case ErrorCodes::CRIT_TEMP:
314  board_error_msg = "Critical Temperature";
315  break;
316  case ErrorCodes::POSCONV:
317  board_error_msg =
318  "Error in SpinTAC Position Convert module";
319  break;
320  case ErrorCodes::POS_ROLLOVER:
321  board_error_msg = "Position Rollover";
322  break;
323  case ErrorCodes::OTHER:
324  board_error_msg = "Other Error";
325  break;
326  default:
327  board_error_msg = "Unknown Error";
328  break;
329  }
330 
331  if (!board_error_msg.empty())
332  {
333  if (!error_msg.empty())
334  {
335  error_msg += " ";
336  }
337 
338  // error of the board with board index to the error message
339  // string
340  error_msg +=
341  "[Board " + std::to_string(i) + "] " + board_error_msg;
342  }
343  }
344  }
345 
346  // check if position is within the limits
347  Vector position = this->joint_modules_.get_measured_angles();
348  if (!config_.is_within_hard_position_limits(position))
349  {
350  if (!error_msg.empty())
351  {
352  error_msg += " | ";
353  }
354 
355  error_msg += "Position limits exceeded.";
356  }
357 
358  return error_msg;
359 }

392 TPL_NJBRD
393 typename NJBRD::Action NJBRD::process_desired_action(
394  const Action &desired_action,
395  const Observation &latest_observation,
396  const double max_torque_Nm,
397  const Vector &safety_kd,
398  const Vector &default_position_control_kp,
399  const Vector &default_position_control_kd,
400  const Vector &lower_position_limits,
401  const Vector &upper_position_limits)
402 {
403  Action processed_action = desired_action;
404 
405  // Position Limits
406  // ---------------
407  // If a joint exceeds the soft position limit, replace the command for that
408  // joint with a position command to the limit
409  for (std::size_t i = 0; i < N_JOINTS; i++)
410  {
411  // Clamp position commands to the allowed range (note that if position
412  // is NaN both conditions are false, so the NaN is preserved).
413  if (processed_action.position[i] < lower_position_limits[i])
414  {
415  processed_action.position[i] = lower_position_limits[i];
416  }
417  else if (processed_action.position[i] > upper_position_limits[i])
418  {
419  processed_action.position[i] = upper_position_limits[i];
420  }
421 
422  auto set_limit_action = [&](double sign, double limit) {
423  // Discard torque command if it pushes further out of the valid
424  // range.
425  if (processed_action.torque[i] * sign > 0)
426  {
427  processed_action.torque[i] = 0;
428  }
429 
430  // If no position is set, set it to the limit value (otherwise it
431  // will already be clamped to the limit range, so it will be fine).
432  if (std::isnan(processed_action.position[i]))
433  {
434  processed_action.position[i] = limit;
435  }
436 
437  // do not allow custom gains
438  processed_action.position_kp[i] = default_position_control_kp[i];
439  processed_action.position_kd[i] = default_position_control_kd[i];
440  };
441 
442  if (latest_observation.position[i] < lower_position_limits[i])
443  {
444  set_limit_action(-1, lower_position_limits[i]);
445  }
446  else if (latest_observation.position[i] > upper_position_limits[i])
447  {
448  set_limit_action(+1, upper_position_limits[i]);
449  }
450  }
451 
452  // Position Controller
453  // -------------------
454 
455  // Run the position controller only if a target position is set for at
456  // least one joint.
457  if (!processed_action.position.array().isNaN().all())
458  {
459  // Replace NaN-values with default gains
460  processed_action.position_kp =
461  processed_action.position_kp.array().isNaN().select(
462  default_position_control_kp, processed_action.position_kp);
463  processed_action.position_kd =
464  processed_action.position_kd.array().isNaN().select(
465  default_position_control_kd, processed_action.position_kd);
466 
467  Vector position_error =
468  processed_action.position - latest_observation.position;
469 
470  // simple PD controller
471  Vector position_control_torque =
472  processed_action.position_kp.cwiseProduct(position_error) -
473  processed_action.position_kd.cwiseProduct(
474  latest_observation.velocity);
475 
476  // position_control_torque contains NaN for joints where target
477  // position is set to NaN! Filter those out and set the torque to
478  // zero instead.
479  position_control_torque =
480  position_control_torque.array().isNaN().select(
481  0, position_control_torque);
482 
483  // Add result of position controller to the torque command
484  processed_action.torque += position_control_torque;
485  }
486 
487  // Safety Checks
488  // -------------
489  // limit to configured maximum torque
490  processed_action.torque =
491  clamp(processed_action.torque, -max_torque_Nm, max_torque_Nm);
492  // velocity damping to prevent too fast movements
493  processed_action.torque -=
494  safety_kd.cwiseProduct(latest_observation.velocity);
495  // after applying checks, make sure we are still below the max. torque
496  processed_action.torque =
497  clamp(processed_action.torque, -max_torque_Nm, max_torque_Nm);
498 
499  return processed_action;
500 }

511 {
512  double start_time_sec = real_time_tools::Timer::get_current_time_sec();
513 
514  Observation observation = get_latest_observation();
515 
516  // Only enable soft position limits once initialization is done (i.e. no

538  joint_modules_.send_torques();
539 
540  real_time_tools::Timer::sleep_until_sec(start_time_sec + 0.001);
541 
542  return applied_action;

546 void NJBRD::_initialize()
547 {
548  joint_modules_.set_position_control_gains(
549  config_.position_control_gains.kp, config_.position_control_gains.kd);
550 
551  bool homing_succeeded = homing(
552  config_.calibration.endstop_search_torques_Nm, config_.home_offset_rad);
553  pause_motors();
554 
555  // NOTE: do not set is_initialized_ yet as we want to allow move_to_position
556  // below to move without position limits (as it might be that after homing
557  // it is out of the limits).
558  if (homing_succeeded)

561 
562  bool reached_goal = false;
563  for (size_t i = 0; i < N_JOINTS; i++)
564  {
565  waypoint[i] = config_.initial_position_rad[i];
566 
567  reached_goal =
568  move_to_position(waypoint,
569  config_.move_to_position_tolerance_rad,
570  config_.calibration.move_steps);
571  }
572  if (!reached_goal)
573  {
574  rt_printf("Failed to reach initial position, timeout exceeded.\n");
575  }
576  }
577 
578  pause_motors();
579 
580  is_initialized_ = homing_succeeded;
581 }

583 TPL_NJBRD
584 bool NJBRD::homing(NJBRD::Vector endstop_search_torques_Nm,
585  NJBRD::Vector home_offset_rad)
586 {
589  const double INDEX_SEARCH_DISTANCE_LIMIT_RAD =
590  (1.5 / motor_parameters_.gear_ratio) * 2 * M_PI;
592  constexpr double INDEX_SEARCH_STEP_SIZE_RAD = 0.0003;
593 
595  Vector travelled_distance = Vector::Zero();
596 
597  rt_printf("Start homing.\n");
598  if (has_endstop_)
599  {
601  constexpr uint32_t MIN_STEPS_MOVE_TO_END_STOP = 1000;
603  constexpr uint32_t SIZE_VELOCITY_WINDOW = 100;
605  constexpr double STOP_VELOCITY = 0.01;
606 
607  static_assert(MIN_STEPS_MOVE_TO_END_STOP > SIZE_VELOCITY_WINDOW,
608  "MIN_STEPS_MOVE_TO_END_STOP has to be bigger than"
609  " SIZE_VELOCITY_WINDOW to ensure correct computation"
610  " of average velocity.");
611 
612  // Move until velocity drops to almost zero (= joints hit the end
613  // stops) but at least for MIN_STEPS_MOVE_TO_END_STOP time steps.
614  // TODO: add timeout to this loop?
615  std::vector<Vector> running_velocities(SIZE_VELOCITY_WINDOW);
616  Vector summed_velocities = Vector::Zero();
617  Vector start_position = get_latest_observation().position;
618  uint32_t step_count = 0;
619  while (step_count < MIN_STEPS_MOVE_TO_END_STOP ||
620  (summed_velocities.maxCoeff() / SIZE_VELOCITY_WINDOW >
621  STOP_VELOCITY))
622  {
623  apply_action_uninitialized(endstop_search_torques_Nm);
624  Vector abs_velocities =
625  get_latest_observation().velocity.cwiseAbs();
626 
627  uint32_t running_index = step_count % SIZE_VELOCITY_WINDOW;
628  if (step_count >= SIZE_VELOCITY_WINDOW)
629  {
630  summed_velocities -= running_velocities[running_index];
631  }
632  running_velocities[running_index] = abs_velocities;
633  summed_velocities += abs_velocities;
634  step_count++;
635 
636 #ifdef VERBOSE
637  Eigen::IOFormat commainitfmt(
638  4, Eigen::DontAlignCols, " ", " ", "", "", "", "");
639  std::cout << ((summed_velocities / SIZE_VELOCITY_WINDOW).array() >
640  STOP_VELOCITY)
641  .format(commainitfmt)
642  << std::endl;
643 #endif
644  }
645  rt_printf("Reached end stop.\n");
646 
647  // compute distance travelled during end-stop search
648  travelled_distance +=
649  get_latest_observation().position - start_position;
650  }
651 
652  // Home on encoder index
653 
654  // Set the search direction for each joint opposite to the end-stop search
655  // direction.
656  Vector index_search_step_sizes;
657  for (unsigned int i = 0; i < N_JOINTS; i++)
658  {
659  index_search_step_sizes[i] = INDEX_SEARCH_STEP_SIZE_RAD;
660  if (endstop_search_torques_Nm[i] > 0)
661  {
662  index_search_step_sizes[i] *= -1;
663  }
664  }
665 
666  HomingReturnCode homing_status =
667  joint_modules_.execute_homing(INDEX_SEARCH_DISTANCE_LIMIT_RAD,
668  home_offset_rad,
669  index_search_step_sizes);
670 
671  rt_printf("Finished homing. Offset between end and start position: ");
672  travelled_distance += joint_modules_.get_distance_travelled_during_homing();
673  for (size_t i = 0; i < N_JOINTS; i++)
674  {
675  // negate the travelled distance so the output can directly be used as
676  // home offset
677  rt_printf("%.3f, ", -travelled_distance[i]);
678  }
679  rt_printf("\n");
680 
681  return homing_status == HomingReturnCode::SUCCEEDED;
682 }

689  // move to the goal position on a minium jerk trajectory, see
690  // https://web.archive.org/web/20200715015252/https://mika-s.github.io/python/control-theory/trajectory-generation/2017/12/06/trajectory-generation-with-a-minimum-jerk-trajectory.html
691 
692  const auto initial_position = get_latest_observation().position;
693 
694  for (int t = 0; t < time_steps; t++)
695  {
696  double alpha = (double)t / (double)time_steps;
697  auto step_goal = initial_position + (goal_pos - initial_position) *
698  (10.0 * std::pow(alpha, 3) -
699  15.0 * std::pow(alpha, 4) +
700  6.0 * std::pow(alpha, 5));
701 
702  apply_action_uninitialized(Action::Position(step_goal));
703  }
704 
705  // check if the goal was really reached
706  const Vector position_error = goal_pos - get_latest_observation().position;
707  bool reached_goal = (position_error.array().abs() < tolerance).all();
708 
709  return reached_goal;
710 }
